select IL.SUMMARY, I.NAME
from ISSUES IL
join ISSUESLISTS I ON I.ID = IL.ISSUESLIST_ID;

select IL.SUMMARY, IL.DESCRIPTION, U.ID as USER_ID
from ISSUES IL
join USERS U ON U.ID = IL.USER_ID_ASSIGNEDTO;

select U.FIRSTNAME, U.LASTNAME, count(*) as NUMBER_OF_TASK
from USERS U
JOIN ISSUES I on U.ID = I.USER_ID_ASSIGNEDTO
group by U.ID;
